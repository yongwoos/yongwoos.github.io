[단지번호붙이기](https://www.acmicpc.net/problem/2667)
```python
import sys
from collections import deque

N = int(sys.stdin.readline().rstrip())
board = [list(map(int, sys.stdin.readline().rstrip())) for _ in range(N)]

deq = deque()

d = [(0,1), (0, -1), (1, 0), (-1, 0)]

visited = [[0]*N for _ in range(N)]

count = 0

total_houses = []

for i in range(N):
    for j in range(N):
        if board[i][j]==1 and visited[i][j]==0:
            houses = 0
            count+=1
            deq.append([i, j])
            visited[i][j] = 1
            while deq:
                x, y = deq.popleft()
                houses += 1
                for dx, dy in d:
                    nx, ny = x+dx, y+dy
                    if 0<=nx<N and 0<=ny<N and visited[nx][ny]==0 and board[nx][ny]==1:
                        board[nx][ny] = count
                        deq.append([nx, ny])
                        visited[nx][ny] = 1
            total_houses.append(houses)

print(count)
answer = sorted(total_houses)

for t in answer:
    print(t)
```
