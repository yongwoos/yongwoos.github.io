[바이러스](https://www.acmicpc.net/problem/2606)

```python
import sys
from collections import deque

computers = int(sys.stdin.readline())
pairs_num = int(sys.stdin.readline())

chain = [[] for _ in range(computers + 1)]

for _ in range(pairs_num):
    a, b = map(int, sys.stdin.readline().split())
    chain[a].append(b)
    chain[b].append(a)

visited = [False] * (computers + 1)

q = deque([1])
visited[1] = True
count = 0

while q:
    node = q.popleft()
    for i in chain[node]:
        if not visited[i]:
            visited[i] = True
            q.append(i)
            count += 1

print(count)
```
